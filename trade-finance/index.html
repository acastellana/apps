How It Works
Six steps. Zero trust required.

The entire deal lifecycle is governed by the smart contract. No intermediaries, no disputes about rates.

01 —
Create Deal
Exporter deploys contract with deal terms: currencies, estimated rate, tolerance, and goods description.
02 —
Fund Escrow
Importer locks USD funds in the smart contract. Money is safe — only released on delivery + rate validation.
03 —
Ship Goods
Exporter submits shipping proof (bill of lading hash). Contract moves to awaiting delivery confirmation.
04 —
Confirm Delivery
Importer confirms receipt of goods. This triggers the AI forex settlement process.
05 —
AI Forex Settlement
GenLayer validators fetch live rates from multiple APIs and reach consensus. Settlement is automatic and trustless.
06 —
Deal Complete
Exact USD amount released to exporter. Difference (if any) returned to importer. All on-chain, all transparent.

The code that runs the deal

@gl.public.write
def settle(self) -> None:
    """AI-powered forex settlement. Called by importer after delivery."""
    assert self.status == "DELIVERY_CONFIRMED", "Wrong status"
    assert gl.message.sender == self.importer, "Only importer"

    # Each validator independently fetches the forex rate
    rate = self._fetch_consensus_rate(
        base=self.invoice_currency,   # "JPY"
        quote=self.settlement_currency  # "USD"
    )

    # Check rate is within agreed tolerance (basis points)
    deviation = abs(rate - self.estimated_rate) / self.estimated_rate
    if deviation > self.tolerance_bps / 10000:
        self.status = "RATE_DISPUTE"
        return

    # Calculate exact settlement amount
    settlement_usd = self.invoice_jpy * rate
    difference = self.escrowed_usd - settlement_usd

    # Release funds
    gl.transfer(self.exporter, settlement_usd)
    if difference > 0:
        gl.transfer(self.importer, difference)  # return excess

    self.final_rate = rate
    self.settlement_usd = settlement_usd
    self.status = "COMPLETED"


@gl.public.write
def _fetch_consensus_rate(self, base: str, quote: str) -> float:
    """Validators each call this independently. GenLayer's optimistic
    execution ensures consensus on the result."""
    result = gl.exec_prompt(
        f"Fetch the current spot exchange rate for {base}/{quote} "
        "from a reliable financial API. Return only the numeric rate."
    )
    return float(result.strip())
@gl.public.write
def resolve_dispute(self) -> None:
    """AI jury evaluates evidence and delivers binding verdict."""
    assert self.status == "DISPUTED", "Not in dispute"

    # Build evidence context for the AI jury
    evidence_ctx = {
        "deal_terms":       self.deal_terms,
        "exporter_evidence": self.exporter_evidence,
        "importer_evidence": self.importer_evidence,
        "shipping_docs":     self.bill_of_lading_hash,
        "dispute_reason":    self.dispute_reason,
        "timeline":          self.get_timeline(),
    }

    # Each validator independently evaluates and votes
    verdict = gl.exec_prompt(
        """You are an impartial trade finance arbitrator.
Review the evidence and return one of:
- EXPORTER_WINS: exporter fulfilled obligations
- IMPORTER_WINS: goods not delivered or misrepresented
- SPLIT:X: split escrowed funds with X% to exporter

Evidence: """ + json.dumps(evidence_ctx),
        output_type=Literal["EXPORTER_WINS", "IMPORTER_WINS", "SPLIT"]
    )

    # Execute verdict automatically
    if verdict == "EXPORTER_WINS":
        gl.transfer(self.exporter, self.escrowed_usd)
    elif verdict == "IMPORTER_WINS":
        gl.transfer(self.importer, self.escrowed_usd)
    else:  # SPLIT
        exporter_share = verdict.split(":")[1]
        gl.transfer(self.exporter, self.escrowed_usd * exporter_share / 100)
        gl.transfer(self.importer, self.escrowed_usd * (1 - exporter_share / 100))

    self.verdict = verdict
    self.status = "RESOLVED"
from genlayer import gl, Literal
import json

class TradeFinanceDeal(gl.Contract):
    """
    Intelligent Trade Finance Contract on GenLayer.
    Handles cross-border deals with AI-validated forex settlement
    and dispute resolution.
    """

    # Deal state
    status: str
    exporter: str
    importer: str
    invoice_jpy: int
    invoice_currency: str
    settlement_currency: str
    estimated_rate: float
    tolerance_bps: int      # e.g. 50 = ±0.5%
    escrowed_usd: float
    bill_of_lading_hash: str
    final_rate: float
    settlement_usd: float

    def __init__(
        self,
        importer: str,
        invoice_jpy: int,
        estimated_rate: float,
        tolerance_bps: int = 50,
        goods_description: str = "",
    ):
        self.exporter = gl.message.sender
        self.importer = importer
        self.invoice_jpy = invoice_jpy
        self.invoice_currency = "JPY"
        self.settlement_currency = "USD"
        self.estimated_rate = estimated_rate
        self.tolerance_bps = tolerance_bps
        self.goods_description = goods_description
        self.status = "PENDING"

    @gl.public.write
    def fund_escrow(self) -> None:
        assert gl.message.sender == self.importer
        assert self.status == "PENDING"
        self.escrowed_usd = gl.message.value  # USD locked in contract
        self.status = "FUNDED"

    @gl.public.write
    def submit_shipping(self, bol_hash: str) -> None:
        assert gl.message.sender == self.exporter
        assert self.status == "FUNDED"
        self.bill_of_lading_hash = bol_hash
        self.status = "SHIPPED"

    @gl.public.write
    def confirm_delivery(self) -> None:
        assert gl.message.sender == self.importer
        assert self.status == "SHIPPED"
        self.status = "DELIVERY_CONFIRMED"
